       ```typescript
       export type PtyEvent =
         | { event: 'data'; data: { bytes: number[] } }
         | { event: 'exit'; data: { code: number | null } };

       export async function ptySpawn(cwd: string, cols: number, rows: number, onEvent: Channel<PtyEvent>): Promise<string> {
         return invoke<string>('pty_spawn', { cwd, cols, rows, onEvent });
       }
       export async function ptyWrite(ptyId: string, data: Uint8Array): Promise<void> {
         return invoke('pty_write', { ptyId, data: Array.from(data) });
       }
       export async function ptyResize(ptyId: string, cols: number, rows: number): Promise<void> {
         return invoke('pty_resize', { ptyId, cols, rows });
       }
       export async function ptyKill(ptyId: string): Promise<void> {
         return invoke('pty_kill', { ptyId });
       }
       // State persistence commands (added below in Task 3)
       export async function stateSave(canvas: CanvasSnapshot): Promise<void> { ... }
       export async function stateLoad(): Promise<CanvasSnapshot | null> { ... }
       ```

    2. **`src/hooks/usePty.ts`:** React hook for PTY lifecycle:
       - Takes `ptyId: string | null`, `termRef: RefObject<Terminal | null>`
       - On mount (when ptyId is set): create `Channel<PtyEvent>`, set `onmessage` handler:
         - `data` event: `termRef.current?.write(new Uint8Array(message.data.bytes))`
         - `exit` event: handle terminal exit (mark as exited, optionally show message)
       - Does NOT call ptySpawn itself -- that happens in the double-click handler via canvasStore
       - Exposes `write(data: string)` -> calls `ptyWrite` with encoded bytes
       - Exposes `resize(cols: number, rows: number)` -> calls `ptyResize`
       - On unmount: calls `ptyKill` if ptyId is set
       - Connect xterm.js `onData` to `write` (user keyboard input -> PTY)

       Wait -- reconsider the hook design. The Channel must be created BEFORE calling ptySpawn (it's a parameter). So the hook should:
       - Accept a callback `onSpawn(channel: Channel<PtyEvent>) => Promise<string>` that the parent calls
       - OR manage the full spawn lifecycle internally, accepting `cwd`, `cols`, `rows` as params

       Better approach: `usePty` manages everything. It returns `{ spawn, write, resize, kill, ptyId, isAlive }`.
       - `spawn(cwd, cols, rows, term: Terminal)`: creates Channel, calls ptySpawn, wires onmessage to term.write, wires term.onData to ptyWrite. Returns ptyId.
       - `write(data)`: sends to ptyWrite
       - `resize(cols, rows)`: sends to ptyResize
       - `kill()`: sends to ptyKill, cleans up
       - Cleanup on unmount: kill if alive
