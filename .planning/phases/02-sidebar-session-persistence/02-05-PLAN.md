1. Add `extensionToTileType` helper to `src/lib/ipc.ts` (or a new `src/lib/fileUtils.ts`):
   ```typescript
   export type ContentTileType = 'note' | 'image' | 'file-preview';

   export function extensionToTileType(ext: string): ContentTileType {
     const lower = ext.toLowerCase();
     if (lower === '.md' || lower === '.mdx') return 'note';
     if (['.png', '.jpg', '.jpeg', '.gif', '.svg', '.webp', '.bmp'].includes(lower)) return 'image';
     return 'file-preview'; // All other files get syntax-highlighted preview
   }
   ```

2. Update `src/components/sidebar/FileTreeItem.tsx`:

   - Add `onDragStart` handler:
     ```typescript
     onDragStart={(e) => {
       if (entry.is_dir) {
         e.preventDefault();
         return;
       }
       const ext = entry.name.includes('.') ? '.' + entry.name.split('.').pop() : '';
       e.dataTransfer.setData('application/excalicode-file', JSON.stringify({
         path: entry.path,
         name: entry.name,
         ext,
       }));
       e.dataTransfer.effectAllowed = 'copy';
     }}
     ```

   - Implementation:
     ```typescript
     addContentNode: (position, tileType, fileData) => {
       const newZIndex = get().maxZIndex + 1;
       const id = crypto.randomUUID();
       // Size based on tile type
       const sizes = {
         'note': { width: 400, height: 300 },
         'image': { width: 400, height: 300 },
         'file-preview': { width: 500, height: 400 },
       };
       const newNode: Node = {
         id,
         type: tileType, // 'note', 'image', or 'file-preview'
         position,
         data: { filePath: fileData.path, fileName: fileData.name },
         dragHandle: '.drag-handle',
         style: sizes[tileType],
         zIndex: newZIndex,
       };
       set((state) => ({
         nodes: [...state.nodes, newNode],
         maxZIndex: newZIndex,
       }));
       forceSave();
     },
     ```
