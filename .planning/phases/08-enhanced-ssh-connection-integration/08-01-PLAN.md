   ```rust
   /// A host entry parsed from ~/.ssh/config.
   #[derive(Debug, Clone, Serialize, Deserialize)]
   pub struct SshConfigHost {
       pub host_alias: String,
       pub hostname: Option<String>,
       pub user: Option<String>,
       pub port: Option<u16>,
       pub identity_file: Option<String>,
   }

   /// Parse ~/.ssh/config and return all non-wildcard Host entries with resolved params.
   ///
   /// Strategy: read the raw file to extract `Host` line aliases (skip wildcards containing `*` or `?`),
   /// then query each alias through ssh2-config for resolved HostName, User, Port, IdentityFile.
   pub fn parse_ssh_config() -> Result<Vec<SshConfigHost>, String> {
       let home = dirs::home_dir().ok_or("No home directory")?;
       let config_path = home.join(".ssh").join("config");
       if !config_path.exists() {
           return Ok(vec![]);
       }

       let raw_content = std::fs::read_to_string(&config_path).map_err(|e| e.to_string())?;

       // Step 1: Extract host aliases from raw file (ssh2-config doesn't enumerate hosts)
       let host_aliases: Vec<String> = raw_content
           .lines()
           .filter_map(|line| {
               let trimmed = line.trim();
               if trimmed.to_lowercase().starts_with("host ") && !trimmed.to_lowercase().starts_with("host *") {
                   let aliases_str = trimmed[5..].trim();
                   // Each Host line can have multiple aliases separated by space
                   Some(
                       aliases_str
                           .split_whitespace()
                           .filter(|a| !a.contains('*') && !a.contains('?'))
                           .map(String::from)
                           .collect::<Vec<_>>()
                   )
               } else {
                   None
               }
           })
           .flatten()
           .collect();

       if host_aliases.is_empty() {
           return Ok(vec![]);
       }

       // Step 2: Parse with ssh2-config for resolved params
       let file = std::fs::File::open(&config_path).map_err(|e| e.to_string())?;
       let mut reader = std::io::BufReader::new(file);
       let config = ssh2_config::SshConfig::default()
           .parse(&mut reader, ssh2_config::ParseRule::ALLOW_UNKNOWN_FIELDS)
           .map_err(|e| format!("Failed to parse SSH config: {}", e))?;

       let mut hosts = Vec::new();
       for alias in host_aliases {
           let params = config.query(&alias);
           hosts.push(SshConfigHost {
               host_alias: alias,
               hostname: params.host_name.clone(),
               user: params.user.clone(),
               port: params.port,
               identity_file: params.identity_file.as_ref().and_then(|files| {
                   files.first().map(|p| p.to_string_lossy().to_string())
               }),
           });
       }

       Ok(hosts)
   }
   ```

3. In `src-tauri/src/ssh/manager.rs`, add `exec_command` method to `SshManager` impl block. IMPORTANT: Clone the handle BEFORE releasing the mutex lock to avoid deadlock (see Pitfall 2 in research):

   ```rust
   /// Execute a command on an active SSH session and return stdout.
   /// Opens a NEW exec channel on the existing connection handle.
   /// The handle is cloned before the lock is released to avoid deadlock.
   pub async fn exec_command(
       &self,
       session_id: &str,
       command: &str,
   ) -> Result<String, Box<dyn std::error::Error + Send + Sync>> {
       // Clone handle outside the lock to avoid holding mutex across await
       let handle = {
           let sessions = self.sessions.lock().await;
           let session = sessions
               .get(session_id)
               .ok_or_else(|| format!("SSH session '{}' not found", session_id))?;
           session.handle.clone()
       };
       // Lock is released here

       // Open a new exec channel (separate from the PTY channel)
       let ch = handle
           .channel_open_session()
           .await
           .map_err(|e| format!("Failed to open exec channel: {}", e))?;

       let (mut read_half, write_half) = ch.split();
       write_half
           .exec(true, command)
           .await
           .map_err(|e| format!("Failed to exec command: {}", e))?;

       let mut output = Vec::new();
       loop {
           match read_half.wait().await {
               Some(russh::ChannelMsg::Data { ref data }) => output.extend_from_slice(data),
               Some(russh::ChannelMsg::Eof) | None => break,
               _ => {}
           }
       }
       Ok(String::from_utf8_lossy(&output).to_string())
   }
   ```

   ```rust
   /// A remote file entry returned from SSH directory listing.
   #[derive(Clone, serde::Serialize)]
   pub struct RemoteFileEntry {
       pub name: String,
       pub is_dir: bool,
   }

   /// List hosts from ~/.ssh/config.
   #[tauri::command]
   pub fn ssh_list_config_hosts() -> Result<Vec<SshConfigHost>, String> {
       parse_ssh_config()
   }

   /// Read a remote directory via SSH exec. Uses `ls -1pA <path>` where directories end with `/`.
   /// Requires an active SSH session (session_id from ssh_connect or a dedicated session).
   #[tauri::command]
   pub async fn ssh_read_remote_dir(
       session_id: String,
       path: String,
       ssh_state: tauri::State<'_, super::manager::SshManager>,
   ) -> Result<Vec<RemoteFileEntry>, String> {
       // Use ls -1pA: -1 = one entry per line, -p = append / to dirs, -A = all except . and ..
       let command = format!("ls -1pA {}", shell_escape(&path));
       let output = ssh_state
           .exec_command(&session_id, &command)
           .await
           .map_err(|e| e.to_string())?;

       let entries: Vec<RemoteFileEntry> = output
           .lines()
           .filter(|line| !line.is_empty())
           .map(|line| {
               if let Some(name) = line.strip_suffix('/') {
                   RemoteFileEntry {
                       name: name.to_string(),
                       is_dir: true,
                   }
               } else {
                   RemoteFileEntry {
                       name: line.to_string(),
                       is_dir: false,
                   }
               }
           })
           .collect();

       Ok(entries)
   }

   /// Open ~/.ssh/config in the system default editor.
   #[tauri::command]
   pub fn ssh_open_config_in_editor() -> Result<(), String> {
       let home = dirs::home_dir().ok_or("No home directory")?;
       let config_path = home.join(".ssh").join("config");

       // Create ~/.ssh/config if it doesn't exist
       if !config_path.exists() {
           if let Some(parent) = config_path.parent() {
               std::fs::create_dir_all(parent).map_err(|e| e.to_string())?;
           }
           std::fs::write(&config_path, "# SSH Config\n").map_err(|e| e.to_string())?;
       }

       open::that(&config_path).map_err(|e| e.to_string())
   }

   /// Minimal shell escaping for remote paths -- wraps in single quotes.
   fn shell_escape(s: &str) -> String {
       format!("'{}'", s.replace('\'', "'\\''"))
   }
   ```
