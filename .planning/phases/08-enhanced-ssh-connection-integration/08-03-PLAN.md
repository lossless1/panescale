```tsx
import { useCallback, useEffect, useState } from "react";
import { sshReadRemoteDir, type RemoteFileEntry } from "../../lib/ipc";
import { useSshStore } from "../../stores/sshStore";

export function RemoteFileTree() {
  const activeRemoteProject = useSshStore((s) => s.activeRemoteProject);
  const clearActiveRemoteProject = useSshStore((s) => s.clearActiveRemoteProject);

  const [expandedDirs, setExpandedDirs] = useState<Set<string>>(new Set());
  const [dirContents, setDirContents] = useState<Map<string, RemoteFileEntry[]>>(new Map());
  const [loading, setLoading] = useState<Set<string>>(new Set());
  const [error, setError] = useState<string | null>(null);

  // Load root directory when remote project changes
  useEffect(() => {
    if (!activeRemoteProject) return;
    setExpandedDirs(new Set());
    setDirContents(new Map());
    setError(null);
    setLoading(new Set(["__root__"]));

    sshReadRemoteDir(activeRemoteProject.sessionId, activeRemoteProject.remotePath)
      .then((entries) => {
        setDirContents(new Map([[activeRemoteProject.remotePath, entries]]));
        setLoading(new Set());
      })
      .catch((err) => {
        setError(String(err));
        setLoading(new Set());
      });
  }, [activeRemoteProject?.sessionId, activeRemoteProject?.remotePath]);

  const toggleDir = useCallback(
    async (fullPath: string) => {
      if (!activeRemoteProject) return;

      if (expandedDirs.has(fullPath)) {
        setExpandedDirs((prev) => {
          const next = new Set(prev);
          next.delete(fullPath);
          return next;
        });
        return;
      }

      setExpandedDirs((prev) => new Set(prev).add(fullPath));

      // Load directory contents if not already loaded
      if (!dirContents.has(fullPath)) {
        setLoading((prev) => new Set(prev).add(fullPath));
        try {
          const entries = await sshReadRemoteDir(activeRemoteProject.sessionId, fullPath);
          setDirContents((prev) => new Map(prev).set(fullPath, entries));
        } catch (err) {
          console.error("Failed to read remote dir:", err);
        } finally {
          setLoading((prev) => {
            const next = new Set(prev);
            next.delete(fullPath);
            return next;
          });
        }
      }
    },
    [activeRemoteProject, expandedDirs, dirContents],
  );

  if (!activeRemoteProject) return null;

  const rootEntries = dirContents.get(activeRemoteProject.remotePath) || [];

  const renderEntry = (entry: RemoteFileEntry, parentPath: string, depth: number) => {
    const fullPath = parentPath === "/" ? `/${entry.name}` : `${parentPath}/${entry.name}`;
    const isExpanded = expandedDirs.has(fullPath);
    const isLoading = loading.has(fullPath);
    const children = dirContents.get(fullPath) || [];

    return (
      <div key={fullPath}>
        <div
          onClick={() => {
            if (entry.is_dir) toggleDir(fullPath);
          }}
          style={{
            display: "flex",
            alignItems: "center",
            gap: 4,
            padding: "3px 8px",
            paddingLeft: 8 + depth * 16,
            fontSize: 12,
            cursor: entry.is_dir ? "pointer" : "default",
            color: "var(--text-primary)",
            whiteSpace: "nowrap",
            overflow: "hidden",
            textOverflow: "ellipsis",
          }}
          onMouseEnter={(e) => {
            (e.currentTarget as HTMLElement).style.backgroundColor = "var(--bg-secondary)";
          }}
          onMouseLeave={(e) => {
            (e.currentTarget as HTMLElement).style.backgroundColor = "transparent";
          }}
        >
          {entry.is_dir && (
            <span style={{ fontSize: 10, width: 10, flexShrink: 0, opacity: 0.6 }}>
              {isLoading ? (
                <span
                  style={{
                    display: "inline-block",
                    width: 10,
                    height: 10,
                    border: "1.5px solid var(--text-secondary)",
                    borderTopColor: "transparent",
                    borderRadius: "50%",
                    animation: "spin 800ms linear infinite",
                  }}
                />
              ) : isExpanded ? (
                "\u25BE"
              ) : (
                "\u25B8"
              )}
            </span>
          )}
          {!entry.is_dir && <span style={{ width: 10, flexShrink: 0 }} />}
          <span style={{ overflow: "hidden", textOverflow: "ellipsis" }}>
            {entry.name}
          </span>
        </div>
        {entry.is_dir && isExpanded && !isLoading && (
          <>
            {children.length === 0 && (
              <div
                style={{
                  paddingLeft: 8 + (depth + 1) * 16,
                  fontSize: 12,
                  fontStyle: "italic",
                  color: "var(--text-secondary)",
                  padding: "3px 8px",
                }}
              >
                Empty directory
              </div>
            )}
            {children.map((child) => renderEntry(child, fullPath, depth + 1))}
          </>
        )}
      </div>
    );
  };

  return (
    <div style={{ flex: 1, overflow: "auto" }}>
      {/* Remote project header */}
      <div
        style={{
          display: "flex",
          alignItems: "center",
          justifyContent: "space-between",
          padding: "4px 8px",
          fontSize: 10,
          color: "var(--text-secondary)",
          borderBottom: "1px solid var(--border)",
          backgroundColor: "var(--bg-sidebar)",
        }}
      >
        <span style={{ overflow: "hidden", textOverflow: "ellipsis", whiteSpace: "nowrap" }}>
          Remote: {activeRemoteProject.user}@{activeRemoteProject.host}:{activeRemoteProject.remotePath}
        </span>
        <button
          onClick={clearActiveRemoteProject}
          title={`Disconnect from ${activeRemoteProject.host}`}
          style={{
            background: "none",
            border: "none",
            cursor: "pointer",
            color: "var(--text-secondary)",
            padding: "2px 4px",
            fontSize: 12,
            borderRadius: 3,
            flexShrink: 0,
          }}
          onMouseEnter={(e) => {
            (e.currentTarget as HTMLElement).style.backgroundColor = "var(--bg-secondary)";
          }}
          onMouseLeave={(e) => {
            (e.currentTarget as HTMLElement).style.backgroundColor = "transparent";
          }}
        >
          &times;
        </button>
      </div>

      {/* Remote file tree area with accent left border */}
      <div style={{ borderLeft: "3px solid rgba(var(--accent-rgb, 99, 102, 241), 0.4)" }}>
        {/* Loading state */}
        {loading.has("__root__") && (
          <div
            style={{
              display: "flex",
              alignItems: "center",
              justifyContent: "center",
              gap: 8,
              padding: 16,
              fontSize: 12,
              color: "var(--text-secondary)",
            }}
          >
            <span
              style={{
                display: "inline-block",
                width: 12,
                height: 12,
                border: "1.5px solid var(--text-secondary)",
                borderTopColor: "transparent",
                borderRadius: "50%",
                animation: "spin 800ms linear infinite",
              }}
            />
            Loading remote files...
          </div>
        )}

        {/* Error state */}
        {error && (
          <div style={{ padding: 16, fontSize: 12, color: "#ef4444" }}>
            Could not list remote directory. Connection may have dropped.{" "}
            <button
              onClick={() => {
                setError(null);
                if (activeRemoteProject) {
                  setLoading(new Set(["__root__"]));
                  sshReadRemoteDir(activeRemoteProject.sessionId, activeRemoteProject.remotePath)
                    .then((entries) => {
                      setDirContents(new Map([[activeRemoteProject.remotePath, entries]]));
                      setLoading(new Set());
                    })
                    .catch((err) => {
                      setError(String(err));
                      setLoading(new Set());
                    });
                }
              }}
              style={{
                background: "none",
                border: "none",
                color: "#ef4444",
                textDecoration: "underline",
                cursor: "pointer",
                fontSize: 12,
                padding: 0,
              }}
            >
              Reconnect?
            </button>
          </div>
        )}

        {/* File entries */}
        {!loading.has("__root__") && !error && rootEntries.length === 0 && (
          <div
            style={{
              padding: 16,
              fontSize: 12,
              fontStyle: "italic",
              color: "var(--text-secondary)",
              textAlign: "center",
            }}
          >
            Empty directory
          </div>
        )}
        {!loading.has("__root__") &&
          !error &&
          rootEntries.map((entry) =>
            renderEntry(entry, activeRemoteProject.remotePath, 0),
          )}
      </div>

      {/* CSS keyframes for spinner */}
      <style>{`
        @keyframes spin {
          to { transform: rotate(360deg); }
        }
      `}</style>
    </div>
  );
}
```
