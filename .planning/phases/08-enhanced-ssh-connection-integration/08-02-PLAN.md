Create `src/components/sidebar/SshQuickConnect.tsx` with the following implementation:

```tsx
import { useEffect, useRef, useState, useCallback } from "react";
import { useSshStore } from "../../stores/sshStore";
import { useCanvasStore } from "../../stores/canvasStore";
import { sshOpenConfigInEditor, type SshConfigHost } from "../../lib/ipc";
import { SshConnectionForm } from "./SshConnectionForm";
import type { SshConnectionConfig } from "../../lib/ipc";

interface SshQuickConnectProps {
  onClose: () => void;
}

export function SshQuickConnect({ onClose }: SshQuickConnectProps) {
  const configHosts = useSshStore((s) => s.configHosts);
  const loadConfigHosts = useSshStore((s) => s.loadConfigHosts);
  const connections = useSshStore((s) => s.connections);
  const addSshTerminalNode = useCanvasStore((s) => s.addSshTerminalNode);
  const dropdownRef = useRef<HTMLDivElement>(null);
  const [showForm, setShowForm] = useState(false);
  const addConnection = useSshStore((s) => s.addConnection);

  // Load config hosts on mount
  useEffect(() => {
    loadConfigHosts();
  }, [loadConfigHosts]);

  // Close on click outside
  useEffect(() => {
    function handleMouseDown(e: MouseEvent) {
      if (dropdownRef.current && !dropdownRef.current.contains(e.target as Node)) {
        onClose();
      }
    }
    document.addEventListener("mousedown", handleMouseDown);
    return () => document.removeEventListener("mousedown", handleMouseDown);
  }, [onClose]);

  // Close on Escape
  useEffect(() => {
    function handleKeyDown(e: KeyboardEvent) {
      if (e.key === "Escape") {
        onClose();
      }
    }
    document.addEventListener("keydown", handleKeyDown);
    return () => document.removeEventListener("keydown", handleKeyDown);
  }, [onClose]);

  const handleConnectSaved = useCallback(
    (conn: SshConnectionConfig) => {
      const viewport = useCanvasStore.getState().viewport;
      const x = -viewport.x + 200;
      const y = -viewport.y + 100;
      addSshTerminalNode({ x, y }, { id: conn.id, host: conn.host, user: conn.user });
      onClose();
    },
    [addSshTerminalNode, onClose],
  );

  const handleConnectConfigHost = useCallback(
    (host: SshConfigHost) => {
      // Create a temporary connection config from the SSH config host
      // and spawn an SSH terminal tile
      const hostname = host.hostname || host.host_alias;
      const user = host.user || "root";
      const port = host.port || 22;
      // Add as a saved connection so ssh_connect can look it up
      const id = crypto.randomUUID();
      addConnection({
        id,
        name: host.host_alias,
        host: hostname,
        port,
        user,
        keyPath: host.identity_file ?? undefined,
      });
      const viewport = useCanvasStore.getState().viewport;
      const x = -viewport.x + 200;
      const y = -viewport.y + 100;
      addSshTerminalNode({ x, y }, { id, host: hostname, user });
      onClose();
    },
    [addConnection, addSshTerminalNode, onClose],
  );

  const handleOpenConfigEditor = useCallback(() => {
    sshOpenConfigInEditor().catch((err) => {
      console.error("Failed to open SSH config:", err);
    });
  }, []);

  const handleSaveNewConnection = useCallback(
    (config: SshConnectionConfig) => {
      addConnection(config);
      setShowForm(false);
    },
    [addConnection],
  );

  const hasConfigHosts = configHosts.length > 0;
  const hasSavedConnections = connections.length > 0;
  const isEmpty = !hasConfigHosts && !hasSavedConnections;

  const sectionHeadingStyle: React.CSSProperties = {
    fontSize: 11,
    fontWeight: 600,
    textTransform: "uppercase",
    letterSpacing: "0.05em",
    color: "var(--text-secondary)",
    padding: "8px 8px 4px 8px",
  };

  const itemStyle: React.CSSProperties = {
    display: "flex",
    alignItems: "baseline",
    gap: 4,
    padding: "8px 8px",
    cursor: "pointer",
    fontSize: 12,
    color: "var(--text-primary)",
    border: "none",
    background: "none",
    width: "100%",
    textAlign: "left",
  };

  return (
    <div
      ref={dropdownRef}
      role="menu"
      style={{
        position: "absolute",
        top: "100%",
        right: 0,
        marginTop: 4,
        backgroundColor: "var(--bg-secondary)",
        border: "1px solid var(--border)",
        borderRadius: 6,
        boxShadow: "0 4px 12px rgba(0,0,0,0.3)",
        minWidth: 240,
        maxWidth: 320,
        zIndex: 9999,
        overflow: "hidden",
      }}
    >
      {isEmpty && (
        <div style={{ padding: 16, fontSize: 12, color: "var(--text-secondary)", textAlign: "center" }}>
          No SSH connections yet. Add hosts to ~/.ssh/config or click '+ New Connection' to get started.
        </div>
      )}

      {hasConfigHosts && (
        <>
          <div style={sectionHeadingStyle}>SSH Config Hosts</div>
          {configHosts.map((host) => (
            <button
              key={host.host_alias}
              role="menuitem"
              onClick={() => handleConnectConfigHost(host)}
              style={itemStyle}
              onMouseEnter={(e) => {
                (e.currentTarget as HTMLElement).style.backgroundColor = "var(--accent)";
                (e.currentTarget as HTMLElement).style.color = "#fff";
              }}
              onMouseLeave={(e) => {
                (e.currentTarget as HTMLElement).style.backgroundColor = "transparent";
                (e.currentTarget as HTMLElement).style.color = "var(--text-primary)";
              }}
            >
              <span style={{ fontWeight: 600 }}>{host.host_alias}</span>
              <span style={{ color: "var(--text-secondary)", fontWeight: 400 }}>
                ({host.hostname || host.host_alias})
              </span>
            </button>
          ))}
        </>
      )}

      {hasSavedConnections && (
        <>
          <div style={sectionHeadingStyle}>Saved Connections</div>
          {connections.map((conn) => (
            <button
              key={conn.id}
              role="menuitem"
              onClick={() => handleConnectSaved(conn)}
              style={itemStyle}
              onMouseEnter={(e) => {
                (e.currentTarget as HTMLElement).style.backgroundColor = "var(--accent)";
                (e.currentTarget as HTMLElement).style.color = "#fff";
              }}
              onMouseLeave={(e) => {
                (e.currentTarget as HTMLElement).style.backgroundColor = "transparent";
                (e.currentTarget as HTMLElement).style.color = "var(--text-primary)";
              }}
            >
              <span style={{ fontWeight: 600 }}>{conn.name}</span>
              <span style={{ color: "var(--text-secondary)", fontWeight: 400 }}>
                ({conn.user}@{conn.host})
              </span>
            </button>
          ))}
        </>
      )}

      {/* Divider */}
      <div style={{ height: 1, backgroundColor: "var(--border)", margin: "4px 8px" }} />

      {/* Actions */}
      <button
        role="menuitem"
        onClick={() => setShowForm(true)}
        style={{
          ...itemStyle,
          color: "var(--text-secondary)",
        }}
        onMouseEnter={(e) => {
          (e.currentTarget as HTMLElement).style.backgroundColor = "var(--accent)";
          (e.currentTarget as HTMLElement).style.color = "#fff";
        }}
        onMouseLeave={(e) => {
          (e.currentTarget as HTMLElement).style.backgroundColor = "transparent";
          (e.currentTarget as HTMLElement).style.color = "var(--text-secondary)";
        }}
      >
        + New Connection...
      </button>
      <button
        role="menuitem"
        onClick={handleOpenConfigEditor}
        style={{
          ...itemStyle,
          color: "var(--text-secondary)",
        }}
        onMouseEnter={(e) => {
          (e.currentTarget as HTMLElement).style.backgroundColor = "var(--accent)";
          (e.currentTarget as HTMLElement).style.color = "#fff";
        }}
        onMouseLeave={(e) => {
          (e.currentTarget as HTMLElement).style.backgroundColor = "transparent";
          (e.currentTarget as HTMLElement).style.color = "var(--text-secondary)";
        }}
      >
        Edit SSH Config
      </button>

      {/* Connection form modal */}
      {showForm && (
        <SshConnectionForm
          onSave={handleSaveNewConnection}
          onCancel={() => setShowForm(false)}
        />
      )}
    </div>
  );
}
```
