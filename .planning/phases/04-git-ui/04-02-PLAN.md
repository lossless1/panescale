export interface GitStatusEntry {

  status: "staged_new" | "staged_modified" | "staged_deleted" | "staged_renamed" | "modified" | "deleted" | "renamed" | "untracked" | "conflicted";

export async function gitStageHunk(repoPath: string, filePath: string, hunkIndex: number): Promise<void>;

- Renders inline unified diff format:

  - If viewing unstaged diff (staged=false): "Stage Hunk" button calls `gitStageHunk(repoPath, filePath, hunkIndex)`.
